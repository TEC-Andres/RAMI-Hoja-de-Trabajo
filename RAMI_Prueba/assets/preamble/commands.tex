\newcommand{\blankpage}{%
    \newpage
    \thispagestyle{empty}%
    \null
    \addtocounter{page}{-1}%
    \newpage%
}

\newcommand{\insertspace}[1]{
    \\\noindent
    \foreach \i in {1,...,#1}{%
        {\makebox[\textwidth]{}}{\ifnum\i=#1\else{\\}\fi}
    }%
}

\newcommand{\longdiv}[2]{%
\[
\begin{array}{r|l}
#1 & #2
\end{array}
\]
}

\newcommand\showdiv[1]{\overline{\smash{\hstretch{.5}{)}\mkern-3.2mu\hstretch{.5}{)}}#1}}

\newcommand{\insertlines}[1]{%

    \noindent
    \foreach \i in {1,...,#1}{%
        {\color{white}aaa}\underline{\makebox[0.9\linewidth]{}}{\ifnum\i=#1\else{\\}\fi}
    }%
}

\newcommand{\boxedtext}[1]{%
    \begin{center}
    \noindent\fbox{%
        \parbox{0.9\textwidth}{%
            #1
        }
    }
    \end{center}
}
